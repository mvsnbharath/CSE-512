def get_weighted_misclass(w,y,yhat):
    return np.sum(w*(np.not_equal(y,yhat)*1.))

def get_misclass(y,yhat):
    return np.sum(np.not_equal(y,yhat))/(len(y)+0.)


def get_exp_loss(y,yhat):
    return np.sum(np.exp(-y*yhat))/(len(y)+0.)
